vue $on, $emit 使用观察者模式

// $on 注册一个事件
vm.$on('test', function (msg) {
  console.log(msg)
})
// $emit 触发 $on 注册的事件, 同时传递参数
vm.$emit('test', 'hi')     // hi

class Observe {
    constructor() {
        this.listeners = {}
    }
    on(eName, listener) {
        this.listeners[eName] = listener
    }
    emit(eName, ...args) {
        const listener = this.listeners[eName]
        if (listener) {
            listener(...args)
        }
    }
}
